+void TimeManager::addEventListener(const std::string& eventName, std::function<void()> callback) {
+    eventListeners_[eventName].push_back(callback);
+}
+
+void TimeManager::removeEventListener(const std::string& eventName) {
+    eventListeners_.erase(eventName);
+}
+
+void TimeManager::dispatchEvent(const std::string& eventName) {
+    auto it = eventListeners_.find(eventName);
+    if (it != eventListeners_.end()) {
+        for (auto& callback : it->second) {
+            callback();
+        }
+    }
+}
+
+bool TimeManager::hasEventListeners(const std::string& eventName) const {
+    auto it = eventListeners_.find(eventName);
+    return it != eventListeners_.end() && !it->second.empty();
+}
