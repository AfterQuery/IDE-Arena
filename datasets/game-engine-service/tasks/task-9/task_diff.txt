+template<typename T>
+T* MemoryPool<T>::allocate() {
+    if (freeList_.empty()) {
+        T* newObj = new T();
+        pool_.push_back(newObj);
+        allocatedCount_++;
+        return newObj;
+    }
+    T* obj = freeList_.back();
+    freeList_.pop_back();
+    allocatedCount_++;
+    return obj;
+}
+
+template<typename T>
+void MemoryPool<T>::deallocate(T* ptr) {
+    if (ptr && contains(ptr)) {
+        freeList_.push_back(ptr);
+        allocatedCount_--;
+    }
+}
+
+template<typename T>
+void MemoryPool<T>::clear() {
+    for (T* ptr : pool_) {
+        delete ptr;
+    }
+    pool_.clear();
+    freeList_.clear();
+    allocatedCount_ = 0;
+}
